4. Typages et API
src/types.ts
export type Category = "tables"|"portes"|"armoires"|"autres";

export type Product = {
  id: string;
  title: string;
  description: string;
  price: number;
  imageUrl: string;
  category: Category;
};

export type CartItem = { productId: string; qty: number };

export type Order = {
  id: string;
  items: CartItem[];
  total: number;
  createdAt: string;
  status: "pending"|"paid"|"cancelled";
};

src/lib/api.ts
import axios from "axios";
export const api = axios.create({
  baseURL: import.meta.env.VITE_API_GATEWAY_URL,
  timeout: 15000,
});

export const Catalogue = {
  list: () => api.get<Product[]>("/catalogue/products"),
  one: (id: string) => api.get<Product>(`/catalogue/products/${id}`),
  search: (q: string) => api.get<Product[]>("/catalogue/products", { params: { q } }),
};

export const CartApi = {
  get: (userId: string) => api.get<CartItem[]>(`/cart/${userId}`),
  add: (userId: string, productId: string, qty=1) => api.post(`/cart/${userId}/items`, { productId, qty }),
  remove: (userId: string, productId: string) => api.delete(`/cart/${userId}/items/${productId}`),
};

export const Orders = {
  create: (userId: string, items: CartItem[]) => api.post<Order>("/orders", { userId, items }),
};

8. Exemple Catalogue (EF1 + EF10)
import { useEffect, useState } from "react";
import { Catalogue } from "../lib/api";
import { loadLocalProducts } from "../lib/catalogueLocal";
import type { Product } from "../types";
import ProductCard from "../components/ProductCard";

export default function CataloguePage() {
  const [items, setItems] = useState<Product[]>([]);
  const [q, setQ] = useState("");

  useEffect(() => {
    (async () => {
      try {
        const { data } = await Catalogue.list();
        setItems(data);
      } catch {
        setItems(await loadLocalProducts());
      }
    })();
  }, []);

  const filtered = items.filter(p =>
    [p.title, p.description, p.category].join(" ").toLowerCase().includes(q.toLowerCase())
  );

  return (
    <section className="container py-10">
      <div className="flex items-center justify-between gap-3 mb-6">
        <h2 className="text-2xl font-semibold">R√©alisations</h2>
        <input
          value={q} onChange={e=>setQ(e.target.value)}
          placeholder="Rechercher (EF10)"
          className="px-3 py-2 rounded-lg bg-white/5 border border-white/10"
        />
      </div>
      <div className="grid sm:grid-cols-2 lg:grid-cols-3 gap-6">
        {filtered.map(p => <ProductCard key={p.id} p={p} />)}
      </div>
    </section>
  );
}
